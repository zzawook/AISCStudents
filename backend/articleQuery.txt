select articleDB.id, articleDB.content,  commentTable.commentList, likedTable.likedList, mediaTable.mediaLink from articleDB left outer join (select commentDB.articleId as CommentArticleId, group_concat(commentDB.order,":",commentDB.content) as commentList from commentDB group by commentDB.articleId) as commentTable on articleDB.id=CommentArticleId left outer join (select likeDB.articleId as LikeArticleId, group_concat(likeDB.writer) as likedList from likeDB group by likeDB.articleId) as likedTable on articleDB.id=LikeArticleId left outer join (select mediaDB.articleId as MediaArticleId, GROUP_CONCAT(mediaDB.mediaOrder,":",mediaDB.mediasrc) as mediaLink from mediaDB group by MediaArticleId) as mediaTable on articleDB.id=MediaArticleId;
